```const express = require('express');
const app = new express();

app.get('/', (req, res) => {
  res.status(200).sendFile(__dirname + '/public/index.html');
});

let port = 8080;
app
.listen(port, () => {
  console.log(
    'time --> ' + new Date(),
    `Your app is listening on port ${port}`
  );
});
```
